# 02 天气查询页

> 学习目标：掌握 AI 编程调用 API 的方法

---

## 需求说明

用 SwiftUI 实现天气查询页面：

### 功能

1. 城市输入框
2. 查询按钮
3. 显示天气信息（温度、天气状况）
4. 加载状态提示
5. 错误处理

### 界面参考

```
┌─────────────────────┐
│  🔍 城市: [北京    ] │
│         [查询天气]   │
├─────────────────────┤
│                     │
│    ☀️ 晴天          │
│    🌡️ 25°C         │
│    💧 湿度: 45%     │
│                     │
└─────────────────────┘
```

### API

使用免费的天气 API：
- 文档：https://openweathermap.org/api
- 示例：`https://api.openweathermap.org/data/2.5/weather?q=Beijing&appid=YOUR_KEY`

---

## 提示

> 遇到困难时的线索

1. 使用 `@State` 管理城市名和天气数据
2. 使用 `async/await` 调用 API
3. 用 `URLSession.shared.data(from:)` 请求数据
4. 解析 JSON 用 `Codable`

## 知识点

### async/await 异步编程

```swift
// 旧写法（Completions）
func fetchWeather(city: String, completion: @escaping (Weather?) -> Void) {
    // ...
}

// 新写法（async/await）
func fetchWeather(city: String) async throws -> Weather {
    let (data, _) = try await URLSession.shared.data(from: url)
    return try JSONDecoder().decode(Weather.self, from: data)
}
```

### Codable 协议

```swift
struct Weather: Codable {
    let main: MainInfo
    let weather: [WeatherInfo]
}

struct MainInfo: Codable {
    let temp: Double
    let humidity: Int
}

struct WeatherInfo: Codable {
    let description: String
    let icon: String
}
```

```swift
import SwiftUI

struct WeatherView: View {
    @State private var city = ""
    @State private var weather: WeatherResponse?
    @State private var isLoading = false
    @State private var errorMessage: String?
    
    var body: some View {
        VStack(spacing: 20) {
            HStack {
                TextField("输入城市名", text: $city)
                    .textFieldStyle(.roundedBorder)
                    .padding(.trailing, 10)
                
                Button("查询") {
                    Task {
                        await fetchWeather()
                    }
                }
                .disabled(city.isEmpty || isLoading)
            }
            .padding(.horizontal)
            
            if isLoading {
                ProgressView("加载中...")
            } else if let error = errorMessage {
                Text("❌ \(error)")
                    .foregroundColor(.red)
            } else if let weather = weather {
                VStack(spacing: 10) {
                    Image(systemName: iconName(for: weather.weather.first?.icon ?? ""))
                        .font(.system(size: 60))
                    
                    Text(weather.weather.first?.description ?? "")
                        .font(.title2)
                    
                    Text("\(Int(weather.main.temp))°C")
                        .font(.system(size: 48, weight: .bold))
                    
                    Text("湿度: \(weather.main.humidity)%")
                        .foregroundColor(.secondary)
                }
            }
            
            Spacer()
        }
        .padding()
    }
    
    private func fetchWeather() async {
        isLoading = true
        errorMessage = nil
        
        // TODO: 替换为你的 API Key
        let apiKey = "YOUR_API_KEY"
        let urlString = "https://api.openweathermap.org/data/2.5/weather?q=\(city)&appid=\(apiKey)&units=metric"
        
        guard let url = URL(string: urlString) else {
            errorMessage = "无效的 URL"
            isLoading = false
            return
        }
        
        do {
            let (data, _) = try await URLSession.shared.data(from: url)
            weather = try JSONDecoder().decode(WeatherResponse.self, from: data)
        } catch {
            errorMessage = "获取天气失败"
        }
        
        isLoading = false
    }
    
    private func iconName(for code: String) -> String {
        switch code {
        case "01": return "sun.max.fill"
        case "02": return "cloud.sun.fill"
        case "03", "04": return "cloud.fill"
        case "09", "10": return "cloud.rain.fill"
        case "11": return "cloud.bolt.fill"
        case "13": return "snow"
        default: return "cloud.fill"
        }
    }
}

struct WeatherResponse: Codable {
    let main: MainInfo
    let weather: [WeatherInfo]
}

struct MainInfo: Codable {
    let temp: Double
    let humidity: Int
}

struct WeatherInfo: Codable {
    let description: String
    let icon: String
}
```
